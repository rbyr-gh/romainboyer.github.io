// Lecture capteur I²C — Arduino
#include <Wire.h>

void setup() {
  Wire.begin();
  Serial.begin(9600);
}

void loop() {
  Wire.requestFrom(0x48, 2);
  int val = Wire.read() << 8 | Wire.read();
  Serial.println(val);
  delay(100);
}
